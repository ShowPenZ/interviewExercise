class LazyManClass {
  constructor(name) {
    const that = this;

    that.name = name;
    that.taskList = []; //Using arrays to store references to functions
    console.log(`Hi I am ${that.name}`);

    // Delay waiting until last execution that.next()，To execute the methods in tasklist;
    setTimeout(() => {
      that.next();
    });
  }

  eat(meal) {
    const that = this;

    const fn = (function(m) {
      return () => {
        console.log(`I am eating ${m}`);

        that.next();
      };
    })(meal);

    that.taskList.push(fn);

    return that;
  }

  sleep(time) {
    const that = this;

    const fn = (function(t) {
      return () => {
        setTimeout(() => {
          console.log(`wait ${t} seconds...`);

          that.next();
        }, t * 1000);
      };
    })(time);

    that.taskList.push(fn);

    return that;
  }

  sleep(time) {
    const that = this;

    const fn = (function(t) {
      return () => {
        setTimeout(() => {
          console.log(`wait ${t} seconds...`);

          that.next();
        }, t * 1000);
      };
    })(time);

    that.taskList.unshift(fn);

    return that;
  }

  next() {
    const that = this;

    const fn = that.taskList.shift();

    fn && fn();
  }
}

function LazyMan(name) {
  return new LazyManClass(name);
}
