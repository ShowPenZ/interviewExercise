给定一个对象为如下

const obj = { 1: 222, 2: 123, 5: 888 };

转换为

[222, 123, null, null, 888, null, null, null, null, null, null, null];

const a = { 1: 222, 2: 123, 5: 888 };
const a1 = { 1: 222, 2: 123, 5: 888, 8: 9898 };
const arr = new Array(12).fill(null);

// 1. 迭代
function dataTransform(obj, arr) {
  let newArr = [];

  arr.reduce((acc, cur, curIndex) => {
    Object.keys(obj).forEach((j, k) => {
      if (!isArray(cur)) {
        // parseInt(j) - 1 === curIndex ----->两个数据结构的关联点
        if (parseInt(j) - 1 === curIndex) {
          cur = obj[parseInt(j)];
        }
      }
    });

    newArr.push(cur);
  }, []);

  return newArr;
}

// 2.循环
function dataTransformByLoop(obj, arr) {
  let newArr = [];

  arr.forEach((d, i) => {
    Object.keys(obj).forEach((j, k) => {
      // parseInt(j) - 1 === curIndex ----->两个数据结构的关联点
      if (parseInt(j) - 1 === i) {
        d = obj[parseInt(j)];
      }
    });

    newArr.push(d);
  });

  return newArr;
}

const x = dataTransform(a, arr);
const y = dataTransform(a1, arr);

const x1 = dataTransformByLoop(a, arr);
const y1 = dataTransformByLoop(a1, arr);

console.log(x);
console.log(y);
console.log(x1);
console.log(y1);
