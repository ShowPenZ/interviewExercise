要求设计 LazyMan 类，实现以下功能。

LazyMan("Tony");
// Hi I am Tony

LazyMan("Tony")
  .sleep(10)
  .eat("lunch");
// Hi I am Tony
// wait 10 seconds...
// I am eating lunch

LazyMan("Tony")
  .eat("lunch")
  .sleep(10)
  .eat("dinner");
// Hi I am Tony
// I am eating lunch
// wait 10 seconds...
// I am eating diner

LazyMan("Tony")
  .eat("lunch")
  .eat("dinner")
  .sleepFirst(5)
  .sleep(10)
  .eat("junk food");

// Hi I am Tony
// wait 5 seconds first...
// I am eating lunch
// I am eating dinner
// wait 10 seconds...
// I am eating junk food

class LazyManClass {
  constructor(name) {
    const that = this;

    that.name = name;
    that.taskList = []; //使用数组来存储任务引用
    console.log(`Hi I am ${that.name}`);

    // 延迟等待最后执行一次that.next()，以执行taskList里面的方法;
    setTimeout(() => {
      that.next();
    });
  }

  eat(meal) {
    const that = this;

    const fn = (function(m) {
      return () => {
        console.log(`I am eating ${m}`);

        that.next();
      };
    })(meal);

    that.taskList.push(fn);

    return that;
  }

  sleep(time) {
    const that = this;

    const fn = (function(t) {
      return () => {
        setTimeout(() => {
          console.log(`wait ${t} seconds...`);

          that.next();
        }, t * 1000);
      };
    })(time);

    that.taskList.push(fn);

    return that;
  }

  sleep(time) {
    const that = this;

    const fn = (function(t) {
      return () => {
        setTimeout(() => {
          console.log(`wait ${t} seconds...`);

          that.next();
        }, t * 1000);
      };
    })(time);

    that.taskList.unshift(fn);

    return that;
  }

  next() {
    const that = this;

    const fn = that.taskList.shift();

    fn && fn();
  }
}

function LazyMan(name) {
  return new LazyManClass(name);
}

LazyMan("Tony")
  .eat("lunch")
  .eat("dinner")
  .sleepFirst(5)
  .sleep(10)
  .eat("junk food");
