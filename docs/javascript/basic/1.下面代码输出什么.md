<h3>下面代码输出什么</h3>

```javascript
var a = 10;
(function() {
  console.log(a);
  a = 5;
  console.log(window.a);
  var a = 20;
  console.log(a);
})();
```

```javascript
// 执行解析步骤：
var a = 10;

(function() {
  // 变量提升（预解析）
  var a = undefined;
  console.log(a);
  // 输出undefined
  a = 5;
  console.log(window.a);
  // 找window(全局)对象的a, 输出10

  a = 20;
  console.log(a);
  // 输出20
})();
```
