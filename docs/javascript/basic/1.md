<h3>What does the following code output</h3>

```javascript
var a = 10;
(function() {
  console.log(a);
  a = 5;
  console.log(window.a);
  var a = 20;
  console.log(a);
})();
```

```javascript
// Perform the parsing steps:
var a = 10;

(function() {
  // Variable lifting
  var a = undefined;
  console.log(a);
  // 输出undefined
  a = 5;
  console.log(window.a);
  // output 10

  a = 20;
  console.log(a);
  // output 20
})();
```
