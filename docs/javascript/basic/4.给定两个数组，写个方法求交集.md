<h3>给定两个数组，写个方法求交集</h3>

```javascript
const a = [1, 2, 2, 3];
const b = [1, 2];
```

```javascript
function intersect(arr1, arr2) {
  let hashMap = {}; // 记录数字对应出现的次数
  let result = [];

  for (let i of arr1) {
    // 已存在hashMap[i]时 相当于数字重复出现的情况下
    if (hashMap[i]) {
      //出现次数加一
      hashMap[i]++;
    } else {
      //第一次出现一次置为1
      hashMap[i] = 1;
    }
  }

  for (let j of arr2) {
    // 在arr2中循环判断是否有与在记录hashmap中相同的数字
    if (hashMap[j] > 0) {
      // 有说明为交集
      result.push(hashMap[j]);
      // 同时减去一次该数字在hashmap记录的次数
      hashMap[j]--;
    }
  }

  return result.sort((a, b) => a - b);
}

intersect(a, b); // [1,2]
```
