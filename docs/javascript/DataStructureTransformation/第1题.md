<h3>给定一个对象为</h3>

```javascript
const obj = [
  { id: 3, parent: 2 },
  { id: 1, parent: null },
  { id: 2, parent: 1 },
];
```

<h3>转化为</h3>

```javascript
const obj = {
  id: 1,
  parent: null,
  child: {
    id: 2,
    parent: 1,
    child: {
      id: 3,
      parent: 2,
    },
  },
};
```

<h3>解题为</h3>

```javascript
const newObj = obj
  .sort((a, b) => b.parent - a.parent)
  .reduce((acc, cur) =>
    // 第一次递归 acc 为 { id: 3, parent: 2 }, cur 为 {id: 2, parent: 1}
    // 所以 第一次递归结果为 { id: 2, parent: 1,child: { id: 3, parent: 2 }}

    // 第二次递归 acc 为 { id: 2, parent: 1,child: { id: 3, parent: 2 }}, cur 为 { id: 1, parent: null },
    // 所以 第二次递归结果为 {id: 1, parent: null,child:{ id: 2, parent: 1,child: { id: 3, parent: 2 }}
    acc ? { ...cur, child: acc } : cur
  );
```
