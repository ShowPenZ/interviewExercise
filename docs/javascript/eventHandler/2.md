<h3>Implement a sleep function</h3>

```javascript
// by promise
const sleep = (time) =>
  new Promise((resolve, reject) => {
    setTimeout(() => {
      resolve(`sleep time is ${time}`);
    }, time);
  });

sleep(2000).then((res) => {
  console.log(res);
});

//Generate
function* sleep(time) {
  yield new Promise((resolve, reject) => {
    setTimeout(() => {
      resolve(`sleep time is ${time}`);
    }, time);
  });
}

sleep(2000)
  .next()
  .value.then((res) => {
    console.log(res);
  });

//async
const sleep = (time) =>
  new Promise((resolve) =>
    setTimeout(() => {
      resolve(`sleep time is ${time}`);
    }, time)
  );

async function output() {
  return await sleep(1000).then((res) => res);
}

output();
```
