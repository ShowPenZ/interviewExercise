<h3>Given two arrays, write a method to find intersection</h3>

```javascript
const a = [1, 2, 2, 3];
const b = [1, 2];
```

```javascript
function intersect(arr1, arr2) {
  let hashMap = {}; // Record the number of times the number corresponds to the occurrence
  let result = [];

  for (let i of arr1) {
    // When a HashMap [i] already exists, it is equivalent to a repeated occurrence of a number
    if (hashMap[i]) {
      //Occurrence times plus one
      hashMap[i]++;
    } else {
      //One occurrence for the first time set to 1
      hashMap[i] = 1;
    }
  }

  for (let j of arr2) {
    // Loop in arr2 to determine if there is the same number as in the record HashMap
    if (hashMap[j] > 0) {
      // If there is one, it means intersection
      result.push(hashMap[j]);
      // Subtract the number of times the number is recorded in HashMap at the same time
      hashMap[j]--;
    }
  }

  return result.sort((a, b) => a - b);
}

intersect(a, b); // [1,2]
```
