<h3>Implement (num1).add(num2).minus(num3)</h3>

```javascript
const isNumber = (num) =>
  Object.prototype.toString.call(num) === "[Object Number]";

Number.prototype.add = function(num) {
  const that = this;

  if (!isNumber(num)) {
    throw new Error("we need a number");
  }

  return that + num;
};

Number.prototype.minus = function(num) {
  const that = this;

  if (!isNumber(num)) {
    throw new Error("we need a number");
  }

  return that - num;
};
```
